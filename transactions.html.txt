    <script>
        const urlParams = new URLSearchParams(window.location.search);
        const account = urlParams.get('account');

        const transactionsList = document.getElementById('transactionsList');
        const accountTitle = document.getElementById('accountTitle');
        accountTitle.textContent = account + ' Transactions';

        let transactions = [];

        if(account === 'Emergency') {
            transactions = [
                { date: '25 Nov 2025', amount: 'R358.21' }
            ];
        } else if(account === 'Lerato Jafta Tsiu') {
            transactions = [];
            const months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov'];
            months.forEach(month => {
                transactions.push({ date: '25 ' + month + ' 2025', amount: 'R100' });
            });
        }

        transactions.forEach(t => {
            const div = document.createElement('div');
            div.className = 'transaction';
            div.textContent = `${t.date} - ${t.amount}`;
            transactionsList.appendChild(div);
        });
    </script>
